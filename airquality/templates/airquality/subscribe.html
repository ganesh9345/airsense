function subscribe() { const email = document.getElementById('emailInput').value.trim(); const city = document.getElementById('subCityInput').value.trim(); const threshold = document.getElementById('thresholdInput').value; if (!email || !city) { showMessage('❌ Please fill all fields!', '#f44336'); return; } document.getElementById('loadingMsg').style.display = 'block'; document.getElementById('subscribeBtn').disabled = true; document.getElementById('subMessage').style.display = 'none'; fetch('/api/subscribe/', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-Requested-With': 'XMLHttpRequest' }, body: JSON.stringify({ email: email, city: city, threshold_aqi: parseFloat(threshold) }) }) .then(res => { if (!res.ok) { throw new Error('Server error: ' + res.status); } return res.json(); }) .then(data => { document.getElementById('loadingMsg').style.display = 'none'; document.getElementById('subscribeBtn').disabled = false; if (data.message) { showMessage('✅ ' + data.message, '#00e676'); document.getElementById('emailInput').value = ''; document.getElementById('subCityInput').value = ''; } else { showMessage('❌ ' + (data.error || 'Something went wrong!'), '#f44336'); } }) .catch(err => { document.getElementById('loadingMsg').style.display = 'none'; document.getElementById('subscribeBtn').disabled = false; showMessage('❌ ' + err.message, '#f44336'); console.error(err); }); }
